# JavaScript 继承

继承的本质就是 复制，即重写原型对象，代之以一个新类型的实例

# 1.原型链继承

构造函数、原型和实例之间的关系： 每个构造函数都有衣蛾原型对象，原型对象都包含一个指向构造函数的指针，而实例都包含一个原型对象的指针

    function SuperType() {
        this.prototype = true
    }
    SuperType.prototype.getSuperType = function() {
        return this.prototype;
    }
    function SubType() {
        this.subPrototype = false
    }
    // 这里是关键，创建SuperType的实例，并将该实例赋值给SubType.prototype
    SubType.prototype = new SuperType()
    SubType.prototype.getSuperType = function() {
        return this.SubPrototype
    }
    var instance = new SubType()
    console.log(instance.getSuperType()) // true

缺点：多个实例对引用类型的操作会被篡改

# 2.借用构造函数继承

使用父类的构造函数来增强子类实例，等同于复制父类的实例给子类（不使用原型）

    function SuperType() {
        this.color = ["red","green","blue"]
    }
    function SubType() {
        // 继承自SuperType
        SupuerType.call(this)
    }

    var instance1 = new SubType()

    instance1.color.push("black")
    console.log(instance1.color) // ["red","green","blue","black"]

    var instance2 = new SubType()
    console.log(instance2.color) // ["red","green","blue"]

核心代码是SuperType.call(this),创建子类的实例时调用SuperType构造函数，于是SuperType的每个实例都会将SuperType中的属性复制一份

缺点：

  • 只能继承父类的实例属性和方法，不能继承原型实现方法
  • 无法实现复用，每个子类都有父类实例函数的副本，影响性能

# 3. 组合继承

组合继承就是将原型链和借用构造函数技术结合到一起，发挥二者长处的一种继承模式。使用原型链实现对原型属性的方法和继承，通过借用构造函数实现对实例属性的继承。这样既能够保证能够通过原型定义的方法实现复用，又能够保证每个实例有自己的属性

组合继承即具有原型链继承能够复用函数的特性，又有借用构造函数方式能够保证每个子类实例能够拥有自己的属性以及向超类传参的特性。但组合继承也并不是完美实现继承的方式，因为这种方式在创建子类是会调用两次超类的构造函数，分别在继承属性和继承原型进行了调用， 会产生两组name和colors属性，一组在实例上 一组在原型上，会造成很大的浪费。

    function Parent(name,age) {
        this.name = name;
        this.age = age;
        this.colors = ['red', 'yellow', 'blue'];
        console.log('我打印了',this.colors); // 最终打印两次
    }
    
    Parent.prototype.getColors = function () {
        console.log('prototype__',this.colors)
    }
    
    function Child(name,age,grade) {
        Parent.call(this, name, age); // 创建子类实例时会执行一次
        this.grade = grade;
    }
    
    Child.prototype = new Parent(); // 指定子类原型会执行一次
    Child.prototype.constructor = Child; // 矫正构造函数
    Child.prototype.sayName = function () {
        console.log(this.name)
    }
    
    let c = new Child('alice',18,50)
    c.sayName()
    c.getColors()

# 4. 原型式继承

这种方法并没有使用严格意义上的构造函数，思想是借助原型可以基于已有的对象创建新对象，同时还不必因此创建自定义类型

    // 在object()函数内部，先创建了一个临时性的构造函数，
    // 然后将传入的对象作为这个构造函数的原型，
    // 最后返回这个临时类型的一个新实例。本质上object()就是完成了一次浅复制操作
    function object(o) {
        function F(){};
        F.prototype = o;
        return new F()
    }
    
    let person = {
        name: 'lat',
        friends: ['leyla', 'court', 'van']
    }
    
    let p1 = object(person);
    p1.name = 'p1';
    p1.friends.push('p1');
    console.log('p1==>', person.friends);
    
    let p2 = object(person);
    p2.name = 'p2';
    p2.friends.push('p2');
    
    console.log('p1.name ==>', p1.name);
    console.log('person.friends ==> ', person.friends);
    
    // ECMAScript5通过新增Object.create()方法规范了原型式继承，
    // 这个方法接受两个参数： 一个用作新对象原型的对象和为新对象定义属性的对象
    
    let p3 = Object.create(person)
    p3.name = 'p3'
    p3.friends.push('p3')
    
    let p4 = Object.create(person);
    p4.name = 'p4';
    p4.friends.push('p4');
    
    console.log('p3.name ==> ', p3.name);
    console.log('Object.create____person.friends ==> ', person.friends)

# 5.寄生式继承

寄生式继承是与原型式继承紧密相关的一种思路，即创建一个仅用于封装继承函数过程的函数，该函数在内部以某种方式来增强对象，最后返回对象

    function object(obj) {
        function F() {};
        F.prototype = obj;
        return new F();
    }
    
    function createAnother(original) {
        let clone = object(original); // 创建新对象
        clone.sayHi = function () {
            console.log('hello world'); // 增强对象，添加属性或方法
        }
        return clone; // 返回对象
    }
    
    const person = {
        name: 'alice',
        friends: ['Sherly', 'Taissy', 'Vant']
    }
    
    let p1 = createAnother(person);
    p1.sayHi(); // hello world

新对象不仅具有person对象的属性和方法，还有自己的 sayHi() 方法 缺陷：使用寄生式继承来为对象添加函数，会由于不能做到***函数复用***造成效率降低，这一点与构造函数模式类似

# 6.寄生组合式继承

基本思路：不必为了指定子类的原型而调用超类的构造函数，我们所需要的无非就是超类原型的一个副本而已。

    // 复制父类的原型对象
    function object(obj) {
        function F(){};
        F.prototype = obj;
        return new F();
    }
    
    // 创建父类的原型副本，改变子类的原型，同事纠正构造函数
    function inherit(subClass, superClass) {
        let parent = object(superClass.prototype);
        parent.constructor = subClass;
        subClass.prototype = parent;
    }

inherit() 函数实现了寄生组合式继承的最简单方式，参数有两个： 子类构造函数和父类构造函数。在函数内部进行了三步操作：

  • 创建超类原型的一个副本
  • 为创建的副本添加constructor属性，从而弥补因重写原型而失去的默认的constructor属性
  • 将新创建的对象赋值给子类的原型。

完整例子：

    // 复制父类的原型对象
    function object(obj) {
        function F(){};
        F.prototype = obj;
        return new F();
    }
    
    // 创建父类的原型副本，改变子类的原型，同事纠正构造函数
    function inherit(subClass, superClass) {
        let parent = object(superClass.prototype);
        parent.constructor = subClass;
        subClass.prototype = parent;
    }
    function Parent(name,age) {
        this.name = name;
        this.age = age;
        this.getName = function () {
            console.log("getName ==>", this.name)
        }
        console.log('调用父类了')
    }

    function Chile(name,age, grade) {
        Parent.call(this, name, age);
        this.grade = grade;
    }

    // 寄生组合的方式
    inherit(Chile, Parent);
    
    let child = new Chile('lucy', 16, 100);
    console.log('child ==> ', child);
    child.getName();

# 7. es6 class继承

es6的继承就是class的继承，通过extends实现继承

    class A {
        constructor(name) {
            this.name = name
        }
    
        hello() {
            console.log(`hello, my name is ${this.name}`)
        }
    }
    
    class B extends A {
        constructor(name, grade) {
            super(name);
            this.grade = grade
        }
    
        myGrade() {
            console.log(`I am at grade ` + this.grade)
        }
    }
    
    let a = new A('xiaoming')
    let b = new B('xiaohong', 100)
    a.hello(); // hello, my name is xiaoming
    b.hello(); // hello, my name is xiaohong
    b.myGrade(); // I am at grade 100
